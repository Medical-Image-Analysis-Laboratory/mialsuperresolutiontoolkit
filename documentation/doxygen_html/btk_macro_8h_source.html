305 inline float RGBtoIndex(float r, float g, float b)
306 {
307  r *= 255; g *= 255; b *= 255;
308 
309  float index = 0.0f;
310 
311  float max = std::max(std::max(r, g), b);
312  float min = std::min(std::min(r, g), b);
313 
314  if(max != min)
315  {
316  if(r == max)
317  {
318  index = ( std::fmod(60.0 * (g-b)/(max-min) + 360.0, 360.0) ) / 360.0;
319  }
320  else if(g == max)
321  {
322  index = ( 60.0 * (b-r)/(max-min) + 120.0 ) / 360.0;
323  }
324  else // b == max
325  {
326  index = ( 60.0 * (r-g)/(max-min) + 240.0 ) / 360.0;
327  }
328  }
329 
330  return index;
331 }
